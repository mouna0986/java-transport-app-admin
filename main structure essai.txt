***App.java (gestion centrale)***
public class App {
    private static App instance;
    private Utilisateur utilisateurConnecte;
    
    public static App getInstance() {
        if (instance == null) {
            instance = new App();
        }
        return instance;
    }
    
    public void run() {
        new MenuPrincipal().afficher();
    }
    
    // Getters/Setters utiles
    public Utilisateur getUtilisateurConnecte() {
        return this.utilisateurConnecte;
    }
}

***MENU PRINCIPALE(aka navigation)
public class MenuPrincipal {
    public void afficher() {
        while (true) {
            System.out.println("\n=== MENU PRINCIPAL ===");
            System.out.println("1. Se connecter");
            System.out.println("2. Quitter");
            
            int choix = ScannerUtils.lireEntier(1, 2);
            
            switch (choix) {
                case 1 -> connexion();
                case 2 -> System.exit(0);
            }
        }
    }
    
    private void connexion() {
        // Appelle l'authentification
        Utilisateur user = AuthService.connecter();
        
        if (user != null) {
            App.getInstance().setUtilisateurConnecte(user);
            
            // Redirige vers le bon menu
            if (user instanceof Admin) {
                new MenuAdmin().afficher();
            } else {
                new MenuUtilisateur().afficher();
            }
        }
    }
}
